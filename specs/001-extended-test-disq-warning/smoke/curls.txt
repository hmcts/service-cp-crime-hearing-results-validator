# -----------------------------------------------------------------------------
# 3.5  DB override disable → rule produces no warnings
# -----------------------------------------------------------------------------
# Three steps: insert override row, wait for cache, repeat 3.1; clean up after.
# Caffeine TTL on the ruleOverrides cache defaults to 30s
# (validation.cache.override-ttl-seconds in CacheConfig), so a 35s sleep is
# safe; bump it if you've configured a longer TTL in this environment.
#
# Note: the validation_rule.severity column is NOT NULL, so when disabling the
# rule we still need to provide a severity — set it to the rule's YAML-default
# (WARNING) which has no semantic effect when enabled=false. (The contract
# document treats the column as semantically optional in this case; the schema
# disagrees.)
# -----------------------------------------------------------------------------

# Step 1 — disable the rule at runtime (works under cpp-dev-environment Postgres)
PGPASSWORD=password123 psql -h localhost -p 5433 -U cpp_user -d results-validator-db -c "
  INSERT INTO validation_rule (id, enabled, severity, updated_at, updated_by)
  VALUES ('DR-DISQ-001', false, 'WARNING', now(), 'manual-smoke-test')
  ON CONFLICT (id) DO UPDATE
    SET enabled=false, severity='WARNING', updated_at=now(), updated_by='manual-smoke-test';
"

# Step 2 — wait for the override-cache TTL, then repeat the 3.1 payload
sleep 35
curl -sX POST http://localhost:8048/api/validation/validate \
  -H "Content-Type: application/json" \
  -H "CJSCPPUID: 7aee5dea-b0de-4604-b49b-86c7788cfc4b" \
  -H "CPP-ACTION: validation-service.validate" \
  -d '{
    "hearingId": "smoke-5",
    "hearingDay": "2026-04-26",
    "courtType": "MAGISTRATES",
    "defendants": [{"id": "d1", "firstName": "Alex", "lastName": "Driver"}],
    "offences": [
      {"id": "o1", "offenceCode": "RT88026", "offenceTitle": "Dangerous driving", "orderIndex": 1}
    ],
    "resultLines": [
      {"id": "r1", "shortCode": "COEW", "label": "Convicted", "defendantId": "d1", "offenceId": "o1"}
    ]
  }' | jq '{errors, warnings, rulesEvaluated}'
# Expect: errors=[], warnings=[], rulesEvaluated still includes both rule ids
# (the rule was evaluated but produced no issue because enabled=false).

Response:
{
  "errors": [],
  "warnings": [],
  "rulesEvaluated": [
    "DR-SENT-002",
    "DR-DISQ-001"
  ]
}

# Step 3 — clean up so the next session is back to default (rule enabled per YAML)
PGPASSWORD=password123 psql -h localhost -p 5433 -U cpp_user -d results-validator-db -c "
  DELETE FROM validation_rule WHERE id='DR-DISQ-001';
"
